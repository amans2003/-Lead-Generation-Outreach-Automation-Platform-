🔁 ZERO-DUPLICATE SYSTEM (Critical — Read Carefully)
This is the most important system in the entire platform. Every day must produce 3,000–3,500 completely fresh leads that have NEVER appeared in the database before — not today, not last month, not ever.
Three-Layer Dedup Architecture
Scraped Raw Lead
      ↓
LAYER 1: PhoneNormalizer.js
  → Strip spaces, dashes, +91, 0091, leading 0
  → Convert to canonical +91XXXXXXXXXX format
  → Reject if not valid 10-digit Indian mobile number
      ↓
LAYER 2: BloomFilterService.js (Redis — microsecond lookup)
  → SHA-256 hash the normalized phone
  → Check bloom filter: "have we EVER seen this hash?"
  → YES → discard immediately (probable duplicate)
  → NO  → proceed to Layer 3
      ↓
LAYER 3: MongoDB SeenLead collection (authoritative check)
  → Query SeenLead by phoneHash
  → EXISTS → discard + increment duplicate counter
  → NOT EXISTS → this is a genuinely new lead!
      ↓
Save to Lead collection + Insert into SeenLead collection
Add hash to Redis Bloom Filter
Emit real-time event via Socket.io

DedupService.js — Core Logic
// services/dedup/DedupService.js
const crypto = require('crypto');

class DedupService {

  /**
   * Check if a phone number has been seen before (across ALL time)
   * @param {string} rawPhone - raw phone from scraper
   * @returns {{ isNew: boolean, normalizedPhone: string, phoneHash: string }}
   */
  async checkPhone(rawPhone) {
    const normalized = PhoneNormalizer.normalize(rawPhone);
    if (!normalized) return { isNew: false }; // invalid number

    const hash = crypto.createHash('sha256').update(normalized).digest('hex');

    // Layer 2: Bloom filter (fast path)
    const inBloom = await BloomFilterService.mightExist(hash);
    if (inBloom) {
      // Bloom filter said "maybe yes" — confirm with DB
      const inDB = await SeenLead.exists({ phoneHash: hash });
      if (inDB) return { isNew: false, normalizedPhone: normalized, phoneHash: hash };
    }

    // Layer 3: Definitive DB check (also catches bloom filter false negatives)
    const inDB = await SeenLead.exists({ phoneHash: hash });
    if (inDB) {
      await BloomFilterService.add(hash); // sync bloom filter
      return { isNew: false, normalizedPhone: normalized, phoneHash: hash };
    }

    return { isNew: true, normalizedPhone: normalized, phoneHash: hash };
  }

  /**
   * Mark a phone as permanently seen — call AFTER successful lead save
   */
  async markAsSeen(phoneHash, phone, source) {
    await Promise.all([
      SeenLead.create({ phoneHash, phone, source }),
      BloomFilterService.add(phoneHash),
    ]);
  }

  /**
   * Bulk check array of raw phones — returns only new ones
   * Used before batch DB inserts for efficiency
   */
  async filterNewPhones(rawPhones) { /* ... */ }
}

PhoneNormalizer.js
// services/dedup/PhoneNormalizer.js
// Handles all Indian number formats:
// 9876543210 / +919876543210 / 09876543210 / 0091-98765-43210 / 98765 43210
function normalize(raw) {
  if (!raw) return null;
  let phone = String(raw).replace(/[\s\-\(\)\.]/g, '');
  if (phone.startsWith('+91')) phone = phone.slice(3);
  else if (phone.startsWith('0091')) phone = phone.slice(4);
  else if (phone.startsWith('91') && phone.length === 12) phone = phone.slice(2);
  else if (phone.startsWith('0') && phone.length === 11) phone = phone.slice(1);
  if (!/^[6-9]\d{9}$/.test(phone)) return null; // invalid Indian mobile
  return `+91${phone}`;
}

    this.browser = await puppeteer.launch({

  async closeBrowser() {
    if (this.browser) await this.browser.close();
  }

const Anthropic = require('@anthropic-ai/sdk');
const client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });

async function generateOutreachMessage(lead, channel) {
  const channelRules = {
    sms: 'Max 160 chars. Casual, Hindi-English mix OK. Include business name. End with CTA.',
    email: 'Professional tone. Subject line + 3-4 sentence body. Clear value proposition.',
    whatsapp: 'Conversational. Use 1-2 relevant emojis. Max 4 lines. Friendly but professional.'
  };

  const response = await client.messages.create({
    model: 'claude-sonnet-4-6',
    max_tokens: 300,
    messages: [{
      role: 'user',
      content: `You are a digital marketing consultant in India reaching out to small businesses.
      
Generate a ${channel} message for:
- Business: ${lead.businessName}
- Category: ${lead.category}  
- City: ${lead.city}
- Channel rules: ${channelRules[channel]}

Our agency offers: Social media management, Google Ads, lead generation, website design for ${lead.category} businesses in India.

Make it feel personal and relevant to their specific business type. NOT generic spam.
Include: "Reply STOP to opt out"
Return ONLY the message text. No explanations.`
    }]
  });

  return response.content[0].text.trim();
}

// services/ai/ResponseClassifier.js
async function classifyResponse(responseText) {
  const response = await client.messages.create({
    model: 'claude-sonnet-4-6',
    max_tokens: 10,
    messages: [{
      role: 'user',
      content: `Classify this business reply as exactly one word:
Reply: "${responseText}"

- "positive" = interested, wants to know more, asked a question, said yes/haan/okay
- "negative" = not interested, stop, remove, no/nahi, angry response
- "neutral" = unclear, out of office, wrong number, gibberish

Respond with ONLY one word: positive, negative, or neutral`
    }]
  });

  const result = response.content[0].text.trim().toLowerCase();
  if (result === 'positive') return 'good_lead';
  if (result === 'negative') return 'not_interested';
  return 'processing';
}

Anti-Bot Evasion
puppeteer-extra-plugin-stealth on all Puppeteer instances
Rotate 50+ real User Agent strings per request
Random delays 2000–5000ms between requests
Rotate free proxies using proxy-chain library
Maintain session cookies per scraper instance
